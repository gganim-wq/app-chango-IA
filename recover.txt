import React, { useState, useEffect, useRef } from 'react';
import { Moon, Sun, Barcode, Sparkles, X, Home, Tag, ShoppingCart, Scan, ShoppingBag, Bot, Trash2, ClipboardList, CheckSquare, Square, Check, Calendar, Package, ChevronLeft, Plus, Mic } from 'lucide-react';
import { CapacitorHttp, Capacitor } from '@capacitor/core';
import { SpeechRecognition as CapacitorSpeech } from '@capacitor-community/speech-recognition';
import CartList from './components/CartList';
import Scanner from './components/Scanner';
import OffersList from './components/OffersList';
import ProductBottomSheet from './components/ProductBottomSheet';
import ManualFallbackModal from './components/ManualFallbackModal';
import { SadBag, BasketPromo } from './components/DesignIcons';
import heroImg from './assets/hero.png';

export default function App() {
  const [cart, setCart] = useState(() => {
    const saved = localStorage.getItem('dia_cart');
    return saved ? JSON.parse(saved) : [];
  });
  const [darkMode, setDarkMode] = useState(() => {
    const saved = localStorage.getItem('dia_dark');
    return saved === 'true' || (!saved && window.matchMedia('(prefers-color-scheme: dark)').matches);
  });
  
  const [activeTab, setActiveTab] = useState('inicio');
  const [isScannerOpen, setIsScannerOpen] = useState(false);
  const [isBottomSheetOpen, setIsBottomSheetOpen] = useState(false);
  const [isManualModalOpen, setIsManualModalOpen] = useState(false);
  const [showClearConfirm, setShowClearConfirm] = useState(false);

  // ── PEDIDOS ──────────────────────────────────────────────────────────
  const [pedidos, setPedidos] = useState(() => {
    const saved = localStorage.getItem('chango_pedidos');
    return saved ? JSON.parse(saved) : [];
  });
  const [showSaveOrderModal, setShowSaveOrderModal] = useState(false);
  const [saveOrderName, setSaveOrderName] = useState('');
  const [selectedPedidoId, setSelectedPedidoId] = useState(null);
  const selectedPedido = pedidos.find(p => p.id === selectedPedidoId) || null;
  const [isCheckMode, setIsCheckMode]   = useState(false);      // modo chequeo activo

  const [scannedProduct, setScannedProduct] = useState(null);
  const [activeEan, setActiveEan] = useState('');
  const [errorType, setErrorType] = useState('not_found');
  const [isLoading, setIsLoading] = useState(false);
  
  const [showAddManual, setShowAddManual] = useState(false);
  const [manualItemText, setManualItemText] = useState('');
  const [isListening, setIsListening] = useState(false);
  const recognitionRef = useRef(null);
  const [pendingVoiceItems, setPendingVoiceItems] = useState(null);

  // ── NEW STATES FOR EMPTY LIST & REPLACEMENT ──
  const [showCreateListModal, setShowCreateListModal] = useState(false);
  const [createListName, setCreateListName] = useState('');
  const [showReplacementModal, setShowReplacementModal] = useState(false);
  const [replacementProduct, setReplacementProduct] = useState(null);

  useEffect(() => { localStorage.setItem('dia_cart',    JSON.stringify(cart));    }, [cart]);
  useEffect(() => { localStorage.setItem('chango_pedidos', JSON.stringify(pedidos)); }, [pedidos]);

  // AUTO-CHEQUEO: Si hay un pedido abierto y se agrega un item al changuito
  useEffect(() => {
    if (selectedPedidoId) {
      const cartEans = new Set(cart.map(item => item.ean));
      let newlyChecked = false;

      setPedidos(prevPedidos => {
        let changed = false;
        const nextPedidos = prevPedidos.map(p => {
          if (p.id !== selectedPedidoId) return p;
          
          const checked = new Set(p.checkedEans || []);
          let pChanged = false;
          p.items.forEach(pedidoItem => {
            if (cartEans.has(pedidoItem.ean) && !checked.has(pedidoItem.ean)) {
              checked.add(pedidoItem.ean);
              pChanged = true;
              newlyChecked = true;
            }
          });
          if (pChanged) {
            changed = true;
            return { ...p, checkedEans: Array.from(checked) };
          }
          return p;
        });
        return changed ? nextPedidos : prevPedidos;
      });

      // Si se auto-marcó algo y no estábamos en modo chequeo, lo activamos para que el usuario vea la magia
      if (newlyChecked && !isCheckMode) {
        setIsCheckMode(true);
      }
    }
  }, [cart, selectedPedidoId, isCheckMode]);

  useEffect(() => {
    if (darkMode) {
      document.documentElement.classList.add('dark');
    } else {
      document.documentElement.classList.remove('dark');
    }
    localStorage.setItem('dia_dark', darkMode.toString());
  }, [darkMode]);

  const getEffectivePrice = (item) => {
    const { listPrice, price, quantity, qtyPromoFormula, tipoOferta } = item;
    if (tipoOferta === 'Por Cantidad' || qtyPromoFormula) {
      if (qtyPromoFormula === '2x1') {
        const payCount = Math.floor(quantity / 2) + (quantity % 2);
        return (payCount * listPrice) / quantity;
      } else if (qtyPromoFormula === '3x2') {
        const payCount = (Math.floor(quantity / 3) * 2) + (quantity % 3);
        return (payCount * listPrice) / quantity;
      } else if (qtyPromoFormula === '4x3') {
        const payCount = (Math.floor(quantity / 4) * 3) + (quantity % 4);
        return (payCount * listPrice) / quantity;
      } else if (qtyPromoFormula === '2da 70') {
        const pairs = Math.floor(quantity / 2);
        return ((pairs * listPrice * 1.3) + ((quantity % 2) * listPrice)) / quantity;
      } else if (qtyPromoFormula === '2da 50') {
        const pairs = Math.floor(quantity / 2);
        return ((pairs * listPrice * 1.5) + ((quantity % 2) * listPrice)) / quantity;
      } else if (qtyPromoFormula === '2da 80') {
        const pairs = Math.floor(quantity / 2);
        return ((pairs * listPrice * 1.2) + ((quantity % 2) * listPrice)) / quantity;
      }
    }
    return (price < listPrice) ? price : listPrice;
  };

  const handleBarcodeScanned = async (ean) => {
    setIsScannerOpen(false);
    setIsLoading(true);
    try {
      const url = `https://diaonline.supermercadosdia.com.ar/api/catalog_system/pub/products/search?fq=alternateIds_Ean:${ean}`;
      const response = await CapacitorHttp.get({
        url,
        headers: {
          'Accept': 'application/json',
          'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36'
        }
      });

      let data = response.data;
      if (!data || data.length === 0) {
        const fallbackUrl = `https://diaonline.supermercadosdia.com.ar/api/catalog_system/pub/products/search?ft=${ean}`;
        const fallbackRes = await CapacitorHttp.get({ url: fallbackUrl, headers: { 'Accept': 'application/json', 'User-Agent': 'Mozilla/5.0' } });
        data = fallbackRes.data;
      }

      setIsLoading(false);

      if (data && data.length > 0) {
        const product = data[0];
        const item = product.items[0];
        const offer = item.sellers[0].commertialOffer;

        let listPrice = offer.ListPrice;
        let price = offer.Price;
        let ofertaActiva = false;
        let tipoOferta = 'Ninguna';
        let qtyPromoFormula = null;
        let promoDetectada = false;

        // ── PRIORIDAD 1: DiscountHighLight ──────────────────────────────────
        // Fuente oficial de Dia: la etiqueta exacta que aparece en la página
        // Ej: [{"<Name>k__BackingField": "40%"}] o [{"<Name>k__BackingField": "3X2"}]
        for (const d of (offer.DiscountHighLight || [])) {
          const n = (d['<Name>k__BackingField'] || d.Name || '').toUpperCase().trim();
          if (!n) continue;
          if (n === '3X2')           { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '3x2';   ofertaActiva = true; promoDetectada = true; break; }
          if (n === '4X3')           { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '4x3';   ofertaActiva = true; promoDetectada = true; break; }
          if (n === '2X1')           { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2x1';   ofertaActiva = true; promoDetectada = true; break; }
          if (n.includes('2DO') || n.includes('2DA') || n.includes('SEGUNDA')) {
            if (n.includes('80')) { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2da 80'; ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('70')) { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2da 70'; ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('50')) { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2da 50'; ofertaActiva = true; promoDetectada = true; break; }
          }
          // Porcentaje directo puro (ej "40%", "30%")
          const m1 = n.match(/^(\d{1,3})%$/);
          if (m1) { tipoOferta = `${m1[1]}%`; ofertaActiva = true; promoDetectada = true; if (listPrice === price) price = listPrice * (1 - parseInt(m1[1]) / 100); break; }
        }

        // ── PRIORIDAD 2: Diferencia matemática de precio ────────────────────
        // Si ListPrice > Price, el descuento ya está aplicado en los precios
        if (!promoDetectada && listPrice > price) {
          tipoOferta = `${Math.round(((listPrice - price) / listPrice) * 100)}%`;
          ofertaActiva = true;
          promoDetectada = true;
        }

        // ── PRIORIDAD 3: PromotionTeasers / Teasers (combos exactos) ────────
        // Solo para combos de cantidad (3x2, 2x1, etc.), nunca para porcentajes
        // ya que los teasers pueden estar presentes aunque no apliquen al precio actual
        if (!promoDetectada) {
          for (const t of [...(offer.PromotionTeasers || []), ...(offer.Teasers || [])]) {
            const n = (t.Name || t['<Name>k__BackingField'] || '').toUpperCase().trim();
            if (!n) continue;
            if (n.includes('3X2'))                                        { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '3x2';   ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('4X3'))                                        { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '4x3';   ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('2X1'))                                        { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2x1';   ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('2DO AL 80') || n.includes('2DA AL 80'))       { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2da 80'; ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('2DO AL 70') || n.includes('2DA AL 70'))       { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2da 70'; ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('2DO AL 50') || n.includes('2DA AL 50'))       { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2da 50'; ofertaActiva = true; promoDetectada = true; break; }
            if (n.includes('CLUB') || n.includes('SOCIO'))                { tipoOferta = 'Club Dia';     ofertaActiva = true; promoDetectada = true; break; }
          }
        }

        // ── PRIORIDAD 4: productClusters — SOLO nombre exactamente un % puro ─
        // Ej: cluster "40%" es válido. "Hasta 2x1 s27" o "2do al 80% s29" son IGNORADOS
        if (!promoDetectada) {
          for (const val of Object.values(product.productClusters || {})) {
            const cv = String(val).toUpperCase().trim();
            if (cv === '2X1') { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '2x1'; ofertaActiva = true; promoDetectada = true; break; }
            if (cv === '3X2') { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '3x2'; ofertaActiva = true; promoDetectada = true; break; }
            if (cv === '4X3') { tipoOferta = 'Por Cantidad'; qtyPromoFormula = '4x3'; ofertaActiva = true; promoDetectada = true; break; }
            const m4 = cv.match(/^(\d{1,3})%$/);  // coincide SOLO con "40%", no con "Hasta 45% s21"
            if (m4) { tipoOferta = `${m4[1]}%`; ofertaActiva = true; promoDetectada = true; if (listPrice === price) price = listPrice * (1 - parseInt(m4[1]) / 100); break; }
          }
        }

        setScannedProduct({
          ean, name: product.productName, listPrice, price,
          tipoOferta, qtyPromoFormula, ofertaActiva, imageUrl: item.images[0]?.imageUrl
        });
        setIsBottomSheetOpen(true);
      } else {
        setActiveEan(ean); setErrorType('not_found'); setIsManualModalOpen(true);
      }
    } catch (err) {
      setIsLoading(false); setActiveEan(ean); setErrorType('timeout'); setIsManualModalOpen(true);
    }
  };

  const handleRemovePedidoItem = (pedidoId, itemEan, e) => {
    if (e) e.stopPropagation();
    setPedidos(prev => prev.map(p => {
      if (p.id !== pedidoId) return p;
      const newItems = p.items.filter(i => i.ean !== itemEan);
      const newChecked = (p.checkedEans || []).filter(ean => ean !== itemEan);
      const tp = newItems.filter(i => !i.isManual).reduce((acc, i) => acc + getEffectivePrice(i) * i.quantity, 0);
      const ts = newItems.filter(i => !i.isManual).reduce((acc, i) => acc + i.listPrice * i.quantity, 0);
      const newTotalItems = newItems.filter(i => !i.isManual).reduce((acc, i) => acc + i.quantity, 0) + newItems.filter(i => i.isManual).length;
      return { 
        ...p, 
        items: newItems, 
        checkedEans: newChecked,
        totalPagar: tp,
        totalAhorro: ts - tp,
        totalItems: newTotalItems
      };
    }));
  };

  const handleAddManualItem = () => {
    if (!manualItemText.trim() || !selectedPedidoId) return;
    const newItem = {
      ean: `manual_${Date.now()}`,
      name: manualItemText.trim(),
      isManual: true,
      quantity: 1,
      listPrice: 0,
      price: 0,
      tipoOferta: null,
      qtyPromoFormula: null,
      ofertaActiva: false,
      imageUrl: null
    };
    setPedidos(prev => prev.map(p => {
      if (p.id !== selectedPedidoId) return p;
      return { ...p, items: [...p.items, newItem], totalItems: p.totalItems + 1 };
    }));
    setManualItemText('');
    setShowAddManual(false);
  };

  const handleVoiceInput = async () => {
    if (isListening) {
      if (Capacitor.isNativePlatform()) {
        await CapacitorSpeech.stop();
      } else if (recognitionRef.current) {
        recognitionRef.current.stop();
      }
      setIsListening(false);
      return;
    }

    setIsListening(true);

    if (Capacitor.isNativePlatform()) {
      try {
        const hasPerm = await CapacitorSpeech.checkPermissions();
        if (hasPerm.speechRecognition !== 'granted') {
          const req = await CapacitorSpeech.requestPermissions();
          if (req.speechRecognition !== 'granted') {
            alert('Permiso de micrófono denegado.');
            setIsListening(false);
            return;
          }
        }

        const result = await CapacitorSpeech.start({
          language: 'es-AR',
          maxResults: 1,
          prompt: "Hablá para dictar tu lista",
          partialResults: false,
          popup: false,
        });

        if (result && result.matches && result.matches.length > 0) {
          const transcript = result.matches[0];
          const rawItems = transcript.split(/\s+y\s+|\s+coma\s+|,/gi).map(s => s.trim()).filter(s => s.length > 1);
          if (rawItems.length > 0) {
            setPendingVoiceItems(rawItems.map(text => text.charAt(0).toUpperCase() + text.slice(1)));
          }
        }
      } catch (err) {
        console.error('Error nativo de voz:', err);
      } finally {
        setIsListening(false);
      }
      return;
    }

    // Web Fallback
    if (!('webkitSpeechRecognition' in window) && !('SpeechRecognition' in window)) {
      alert('Tu dispositivo no soporta el dictado por voz de forma nativa.');
      setIsListening(false);
      return;
    }
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    const recognition = new SpeechRecognition();
    recognitionRef.current = recognition;
    recognition.lang = 'es-AR';
    recognition.continuous = false;
    recognition.interimResults = false;

    recognition.onresult = (event) => {
      const transcript = event.results[0][0].transcript;
      const rawItems = transcript.split(/\s+y\s+|\s+coma\s+|,/gi).map(s => s.trim()).filter(s => s.length > 1);
      
      if (rawItems.length > 0) {
        setPendingVoiceItems(rawItems.map(text => text.charAt(0).toUpperCase() + text.slice(1)));
      }
    };

    recognition.onerror = (event) => {
      console.error('Error en reconocimiento de voz:', event.error);
      setIsListening(false);
    };

    recognition.onend = () => setIsListening(false);

    recognition.start();
  };

  const confirmVoiceItems = () => {
    if (!pendingVoiceItems || pendingVoiceItems.length === 0) {
      setPendingVoiceItems(null);
      return;
    }
    
    setPedidos(prev => prev.map(p => {
      if (p.id !== selectedPedidoId) return p;
      const newItems = pendingVoiceItems.map((text, idx) => ({
        ean: `manual_${Date.now()}_${idx}`,
        name: text,
        isManual: true,
        quantity: 1,
        listPrice: 0,
        price: 0,
        tipoOferta: null,
        qtyPromoFormula: null,
        ofertaActiva: false,
        imageUrl: null
      }));
      return { ...p, items: [...p.items, ...newItems], totalItems: p.totalItems + newItems.length };
    }));
    setPendingVoiceItems(null);
  };

  const removePendingVoiceItem = (index) => {
    setPendingVoiceItems(prev => prev.filter((_, i) => i !== index));
  };

  const handleCreateEmptyList = () => {
    if (!createListName.trim()) return;
    const newOrder = {
      id: Date.now(),
      name: createListName.trim(),
      date: new Date().toISOString(),
      items: [],
      checkedEans: [],
      totalPagar: 0,
      totalAhorro: 0,
      totalItems: 0,
    };
    setPedidos(prev => [newOrder, ...prev]);
    setCreateListName('');
    setShowCreateListModal(false);
  };

  const handleConfirmProduct = (product) => {
    if (selectedPedido) {
      const checkedSet = new Set(selectedPedido.checkedEans || []);
      const pendingManuals = selectedPedido.items.filter(i => i.isManual && !checkedSet.has(i.ean));
      
      if (pendingManuals.length > 0) {
        setReplacementProduct(product);
        setIsBottomSheetOpen(false);
        setShowReplacementModal(true);
        return;
      }
    }
    proceedConfirmProduct(product);
  };

  const handleReplacementSelect = (manualItemEan) => {
    if (manualItemEan) {
      setPedidos(prev => prev.map(p => {
        if (p.id !== selectedPedidoId) return p;
        const newItems = p.items.map(item => {
          if (item.ean === manualItemEan) {
            return {
              ...item,
              ean: replacementProduct.ean,
              name: replacementProduct.name,
              isManual: false,
              listPrice: replacementProduct.listPrice,
              price: replacementProduct.price,
              tipoOferta: replacementProduct.tipoOferta,
              qtyPromoFormula: replacementProduct.qtyPromoFormula,
              ofertaActiva: replacementProduct.ofertaActiva,
              imageUrl: replacementProduct.imageUrl
            };
          }
          return item;
        });
        return { ...p, items: newItems };
      }));
    }
    proceedConfirmProduct(replacementProduct, true);
    setShowReplacementModal(false);
    setReplacementProduct(null);
  };

  const proceedConfirmProduct = (product, isReplacement = false) => {
    setCart((prevCart) => {
      const existingIndex = prevCart.findIndex(item => item.ean === product.ean && item.qtyPromoFormula === product.qtyPromoFormula);
      if (existingIndex > -1) {
        const newCart = [...prevCart];
        newCart[existingIndex].quantity += product.quantity;
        return newCart;
      }
      return [...prevCart, product];
    });
    setIsBottomSheetOpen(false);
    
    if (selectedPedido) {
      const isProductInPedido = isReplacement || selectedPedido.items.some(i => i.ean === product.ean);
      if (isProductInPedido) {
        setActiveTab('pedidos');
      } else {
        setActiveTab('changuito');
      }
    } else {
      setActiveTab('changuito');
    }
  };

  const handleSaveOrder = () => {
    if (!saveOrderName.trim()) return;
    const tp = cart.reduce((acc, i) => acc + getEffectivePrice(i) * i.quantity, 0);
    const ts = cart.reduce((acc, i) => acc + i.listPrice * i.quantity, 0);
    const newOrder = {
      id: Date.now(),
      name: saveOrderName.trim(),
      date: new Date().toISOString(),
      items: [...cart],
      checkedEans: [],
      totalPagar: tp,
      totalAhorro: ts - tp,
      totalItems: cart.reduce((acc, i) => acc + i.quantity, 0),
    };
    setPedidos(prev => [newOrder, ...prev]);
    setSaveOrderName('');
    setShowSaveOrderModal(false);
    setCart([]);
    setActiveTab('pedidos');
  };

  const totalPagar = cart.reduce((acc, i) => acc + getEffectivePrice(i) * i.quantity, 0);
  const totalSinDescuento = cart.reduce((acc, i) => acc + i.listPrice * i.quantity, 0);
  const totalAhorro = totalSinDescuento - totalPagar;
  const totalItems = cart.reduce((acc, i) => acc + i.quantity, 0);
  const formatCurrency = (val) => new Intl.NumberFormat('es-AR', { style: 'currency', currency: 'ARS' }).format(val);

  return (
    <div className="min-h-screen bg-figmaBlack flex flex-col items-center justify-center p-0 transition-colors duration-300 overflow-hidden font-sans">

      <div className="w-full max-w-[440px] h-screen bg-figmaBlack flex flex-col relative animate-fade-in">

        {/* HEADER FIGMA STYLE */}
        <header className="pt-10 pb-4 px-8 flex items-center shrink-0">
          <div className="flex items-center gap-1">
            <span className="text-white font-black text-2xl tracking-tighter">
              {activeTab === 'changuito' ? 'Mi Changuito ' : 'Chango '}
            </span>
            <span className="text-figmaPurple font-black text-2xl tracking-tighter">IA</span>
          </div>
        </header>

        <main className="flex-1 overflow-y-auto px-6 no-scrollbar pb-32">

          {activeTab === 'inicio' && (
            <div className="flex flex-col h-full animate-fade-in pt-4">
              {/* BANNER IA CON ROBOT */}
              <div className="bg-[#1A1D24] rounded-[2.5rem] p-6 relative overflow-hidden border border-white/5 shadow-2xl flex items-center gap-4 mb-10">
                 <div className="flex-1 z-10 space-y-2">
                    <h2 className="text-[13px] font-bold text-white leading-tight">
                        Chango IA: Tu asistente de ahorro. Escanea productos y no te olvides de nada.
                    </h2>
                 </div>
                 <div className="w-16 h-16 relative z-10 bg-white/5 rounded-2xl flex items-center justify-center overflow-hidden">
                    <Bot className="w-10 h-10 text-white opacity-80" />
                 </div>
              </div>

              <div className="flex-1 flex flex-col items-center justify-center space-y-8 text-center pb-20">
                  <div className="w-32 h-32 bg-white/5 rounded-3xl flex items-center justify-center border border-white/10">
                      <SadBag className="w-20 h-20 opacity-60" />
                  </div>
                  <div className="space-y-3 px-4">
                      <h3 className="text-xl font-bold text-white">Tu changuito está vacío</h3>
                      <p className="text-[12px] text-gray-500 font-medium leading-relaxed px-8">
                          Escanea los códigos de barra de tus productos favoritos para cargar tu carrito. Verifica tu Changuito para ver las Ofertas. Consulta tu lista de Pedidos.
                      </p>
                  </div>
              </div>

              {/* BOTÓN ESCANEAR AL FINAL (FIGMA 4) */}
              <div className="pb-10">
                  <button
                      onClick={() => setIsScannerOpen(true)}
                      className="w-full py-5 bg-figmaRed text-white rounded-full font-black text-sm uppercase tracking-widest flex items-center justify-center gap-3 shadow-accent"
                  >
                      <Scan className="w-5 h-5" />
                      ESCANEAR
                  </button>
              </div>
            </div>
          )}

          {activeTab === 'changuito' && (
            <div className="pt-2 space-y-6 animate-fade-in relative">
                {/* CABECERA FIJA XHDPI STYLE (Imagen 1) */}
                <div className="space-y-6">
                    <div className="grid grid-cols-2 gap-4">
                        <div className="bg-figmaBlue p-6 rounded-[2rem] shadow-xl relative overflow-hidden h-32 flex flex-col justify-between">
                            <p className="text-[10px] font-bold text-blue-100 uppercase tracking-widest">Total a Pagar:</p>
                            <p className="text-xl font-black text-white">{formatCurrency(totalPagar)}</p>
                        </div>
                        <div className="bg-gradient-to-br from-figmaCyan to-figmaPurple p-6 rounded-[2rem] shadow-xl relative overflow-hidden h-32 flex flex-col justify-between">
                            <div className="flex justify-between items-start">
                                <p className="text-[10px] font-bold text-white uppercase tracking-widest">Tu Ahorro:</p>
                                <div className="bg-white/20 p-1.5 rounded-lg">
                                    <Sparkles className="w-4 h-4 text-white" />
                                </div>
                            </div>
                            <p className="text-xl font-black text-white">{formatCurrency(totalAhorro)}</p>
                        </div>
                    </div>

                    <div className="grid grid-cols-2 gap-4">
                        <button
                          onClick={() => { setSaveOrderName(''); setShowSaveOrderModal(true); }}
                          className="py-4 bg-figmaRed text-white rounded-full font-black text-xs uppercase tracking-widest active:scale-95 transition-all"
                        >
                            Finalizar
                        </button>
                        <button onClick={() => setActiveTab('inicio')} className="py-4 border-2 border-white/20 text-white rounded-full font-black text-xs uppercase tracking-widest bg-transparent">
                            Seguir
                        </button>
                    </div>

                    {/* BOTÓN VACIAR CARRITO */}
                    {cart.length > 0 && (
                      <button
                        onClick={() => setShowClearConfirm(true)}
                        className="w-full py-3.5 flex items-center justify-center gap-2 border border-red-900/40 text-red-500 hover:text-red-400 hover:border-red-700/60 bg-red-950/20 rounded-full font-black text-[11px] uppercase tracking-[0.2em] active:scale-95 transition-all"
                      >
                        <Trash2 className="w-3.5 h-3.5" />
                        Vaciar Carrito
                      </button>
                    )}
                </div>

                <div className="pt-4">
                    <CartList
                        items={cart}
                        getEffectivePrice={getEffectivePrice}
                        onUpdateQuantity={(ean, q) => setCart(prev => prev.map(i => i.ean === ean ? {...i, quantity: q} : i))}
                        onRemoveItem={(ean) => setCart(prev => prev.filter(i => i.ean !== ean))}
                        onClearCart={() => setShowClearConfirm(true)}
                    />
                </div>
            </div>
          )}

          {activeTab === 'ofertas' && <div className="pt-4"><OffersList onSelectOffer={handleBarcodeScanned} /></div>}

          {/* ── TAB PEDIDOS ────────────────────────────────────────────── */}
          {activeTab === 'pedidos' && (
            selectedPedido ? (() => {
              const fmtP = (v) => new Intl.NumberFormat('es-AR',{style:'currency',currency:'ARS'}).format(v);
              const checkedSet = new Set(selectedPedido.checkedEans || []);
              const allChecked = selectedPedido.items.every(i => checkedSet.has(i.ean));
              return (
                <div className="flex flex-col animate-fade-in space-y-4 pt-2">
                  {/* HEADER */}
                  <div className="flex items-center justify-between">
                    <div>
                      <p className="text-[9px] font-bold text-gray-600 uppercase tracking-widest">Pedido de</p>
                      <h3 className="text-xl font-black text-white">{selectedPedido.name}</h3>
                    </div>
                    <button onClick={() => { setSelectedPedidoId(null); setIsCheckMode(false); }} className="w-10 h-10 rounded-2xl bg-white/5 flex items-center justify-center text-gray-400 active:scale-95 transition-all">
                      <ChevronLeft className="w-5 h-5" />
                    </button>
                  </div>

                  {/* RESUMEN */}
                  <div className="grid grid-cols-2 gap-3">
                    <div className="bg-figmaBlue/15 border border-figmaBlue/20 rounded-2xl p-4">
                      <p className="text-[9px] font-bold text-blue-400 uppercase tracking-widest">Total</p>
                      <p className="text-base font-black text-white">{fmtP(selectedPedido.totalPagar)}</p>
                    </div>
                    <div className="bg-green-900/20 border border-green-900/30 rounded-2xl p-4">
                      <p className="text-[9px] font-bold text-green-400 uppercase tracking-widest">Ahorro</p>
                      <p className="text-base font-black text-white">{fmtP(selectedPedido.totalAhorro)}</p>
                    </div>
                  </div>

                  {/* LISTA DE ITEMS */}
                  <div className="space-y-3 pb-2">
                    {[...selectedPedido.items].sort((a, b) => {
                      const aChecked = checkedSet.has(a.ean);
                      const bChecked = checkedSet.has(b.ean);
                      if (aChecked && !bChecked) return -1;
                      if (!aChecked && bChecked) return 1;
                      return 0;
                    }).map(item => {
                      const checked = checkedSet.has(item.ean);
                      return (
                        <div
                          key={item.ean}
                          onClick={() => {
                            if (!isCheckMode) return;
                            setPedidos(prev => prev.map(p => {
                              if (p.id !== selectedPedidoId) return p;
                              const c = new Set(p.checkedEans || []);
                              checked ? c.delete(item.ean) : c.add(item.ean);
                              return { ...p, checkedEans: Array.from(c) };
                            }));
                          }}
                          className={`flex items-center gap-4 p-4 rounded-2xl border transition-all ${
                            isCheckMode ? 'cursor-pointer active:scale-[0.98]' : ''
                          } ${
                            checked ? 'bg-green-900/20 border-green-800/30 opacity-60' : 'bg-[#1A1D24] border-white/5'
                          }`}
                        >
                          {/* Checkbox */}
                          {isCheckMode && (
                            <div className={`shrink-0 w-6 h-6 rounded-lg flex items-center justify-center transition-all ${
                              checked ? 'bg-green-500 text-white' : 'bg-white/5 border border-white/20'
                            }`}>
                              {checked && <Check className="w-3.5 h-3.5" />}
                            </div>
                          )}
                          {/* Imagen */}
                          <div className={`w-12 h-12 rounded-xl overflow-hidden flex items-center justify-center shrink-0 p-1 ${item.isManual ? 'bg-figmaPurple/20' : 'bg-white'}`}>
                            {item.imageUrl
                              ? <img src={item.imageUrl} alt="" className="w-full h-full object-contain" />
                              : (item.isManual ? <ClipboardList className="w-5 h-5 text-figmaPurple" /> : <Package className="w-6 h-6 text-gray-300" />)}
                          </div>
                          {/* Info */}
                          <div className="flex-1 min-w-0">
                            <p className={`text-xs font-black truncate ${checked ? 'line-through text-gray-500' : 'text-white'}`}>{item.name}</p>
                            {!item.isManual && <p className="text-[10px] text-gray-600 mt-0.5">x{item.quantity}</p>}
                          </div>
                          {/* Precio / Delete */}
                          <div className="flex items-center gap-3 shrink-0">
                            {!item.isManual && (
                              <p className="text-sm font-black text-white">{fmtP(getEffectivePrice(item) * item.quantity)}</p>
                            )}
                            {!isCheckMode && (
                              <button 
                                onClick={(e) => handleRemovePedidoItem(selectedPedidoId, item.ean, e)} 
                                className="p-1.5 text-gray-500 hover:text-figmaRed active:scale-95 transition-all bg-white/5 rounded-lg"
                              >
                                <Trash2 className="w-4 h-4" />
                              </button>
                            )}
                          </div>
                        </div>
                      );
                    })}
                    
                    {/* AGREGAR MANUAL */}
                    {showAddManual ? (
                      <div className="flex items-center gap-3 p-3 rounded-2xl bg-[#1A1D24] border border-figmaPurple/30 animate-fade-in">
                        <input
                          autoFocus
                          type="text"
                          value={manualItemText}
                          onChange={(e) => setManualItemText(e.target.value)}
                          placeholder="Ej: Ir a la verdulería..."
                          className="flex-1 bg-transparent text-white text-sm outline-none placeholder-gray-500"
                          onKeyDown={(e) => e.key === 'Enter' && handleAddManualItem()}
                        />
                        <button onClick={() => { setShowAddManual(false); setManualItemText(''); }} className="p-2 text-gray-400">
                          <X className="w-4 h-4" />
                        </button>
                        <button onClick={handleAddManualItem} className="p-2 bg-figmaPurple text-white rounded-xl">
                          <Check className="w-4 h-4" />
                        </button>
                      </div>
                    ) : (
                      <div className="flex gap-2">
                        <button
                          onClick={() => setShowAddManual(true)}
                          className="flex-1 py-4 rounded-2xl border border-dashed border-white/20 text-gray-400 flex items-center justify-center gap-2 active:scale-95 transition-all text-sm font-bold"
                        >
                          <span className="text-lg leading-none mt-[-2px]">+</span> Agregar Manual
                        </button>
                        <button
                          onClick={handleVoiceInput}
                          className={`w-[60px] flex items-center justify-center rounded-2xl active:scale-95 transition-all border ${
                            isListening ? 'bg-figmaRed border-red-500 text-white animate-pulse' : 'bg-white/5 border-dashed border-white/20 text-gray-400'
                          }`}
                        >
                          <Mic className={`w-5 h-5 ${isListening ? 'animate-bounce' : ''}`} />
                        </button>
                      </div>
                    )}
                  </div>

                  {/* FOOTER: CHEQUEAR PEDIDO */}
                  <div className="flex flex-col gap-3 pt-2">
                    {isCheckMode && (
                      <div className="flex justify-between items-center px-2">
                        <p className="text-center text-[10px] font-bold text-gray-500 uppercase tracking-widest">
                          {checkedSet.size} / {selectedPedido.items.length} products chequeados
                        </p>
                        <button onClick={() => setPedidos(prev => prev.map(p => p.id === selectedPedidoId ? {...p, checkedEans: []} : p))} className="text-[10px] font-bold text-red-400 uppercase tracking-widest">
                          Reiniciar
                        </button>
                      </div>
                    )}
                    <button
                      onClick={() => setIsCheckMode(v => !v)}
                      className={`w-full py-4 rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all flex items-center justify-center gap-2 ${
                        isCheckMode
                          ? 'bg-green-600 text-white shadow-lg shadow-green-900/30'
                          : 'bg-figmaRed text-white shadow-lg shadow-red-900/30'
                      }`}
                    >
                      {isCheckMode ? <><CheckSquare className="w-4 h-4" /> Finalizar Chequeo</> : <><Square className="w-4 h-4" /> Chequear Pedido</>}
                    </button>
                    <button
                      onClick={() => { setPedidos(prev => prev.filter(p => p.id !== selectedPedidoId)); setSelectedPedidoId(null); }}
                      className="w-full py-3 bg-white/5 text-red-500/70 rounded-2xl font-black text-[11px] uppercase tracking-widest active:scale-95 transition-all border border-white/5"
                    >
                      Eliminar Pedido
                    </button>
                  </div>
                </div>
              );
            })() : (
              <div className="pt-2 space-y-4 animate-fade-in">
                <div className="flex items-center justify-between">
                  <h2 className="text-base font-black text-white">Pedidos Guardados</h2>
                  <span className="text-[10px] font-black uppercase tracking-widest text-gray-500 bg-white/5 px-3 py-1 rounded-full border border-white/5">
                    {pedidos.length} pedido{pedidos.length !== 1 ? 's' : ''}
                  </span>
                </div>

                <button
                  onClick={() => { setCreateListName(''); setShowCreateListModal(true); }}
                  className="w-full py-4 bg-figmaPurple text-white rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all shadow-lg shadow-purple-900/30 flex items-center justify-center gap-2 mb-2"
                >
                  <Plus className="w-5 h-5" />
                  Nueva Lista Vacía
                </button>

                {pedidos.length === 0 ? (
                  <div className="flex flex-col items-center justify-center gap-4 py-20 text-center">
                    <div className="w-20 h-20 rounded-[2rem] bg-white/5 border border-white/5 flex items-center justify-center">
                      <ClipboardList className="w-9 h-9 text-gray-700" />
                    </div>
                    <p className="text-sm font-bold text-gray-600">Todavía no guardaste ningún pedido</p>
                    <p className="text-xs text-gray-700 max-w-[220px] leading-relaxed">Finalizá un changuito y asignale un nombre para guardarlo acá</p>
                  </div>
                ) : (
                  <div className="space-y-3 pb-6">
                    {pedidos.map(p => {
                      const checkedCount = p.checkedEans?.length || 0;
                      const totalCount = p.items.length;
                      const progress = totalCount === 0 ? 0 : Math.round((checkedCount / totalCount) * 100);
                      const isComplete = checkedCount === totalCount && totalCount > 0;
                      
                      return (
                      <button
                        key={p.id}
                        onClick={() => { setSelectedPedidoId(p.id); setIsCheckMode(false); }}
                        className="w-full bg-[#1A1D24] rounded-3xl p-5 border border-white/5 flex flex-col gap-4 active:scale-[0.98] transition-all text-left"
                      >
                        <div className="flex items-center gap-4 w-full">
                          <div className={`w-12 h-12 rounded-2xl flex items-center justify-center shrink-0 border transition-all ${
                            isComplete ? 'bg-green-500/10 border-green-500/20' : 'bg-figmaPurple/20 border-figmaPurple/20'
                          }`}>
                            <Package className={`w-5 h-5 ${isComplete ? 'text-green-500' : 'text-figmaPurple'}`} />
                          </div>
                          <div className="flex-1 min-w-0">
                            <p className="text-sm font-black text-white truncate">{p.name}</p>
                            <p className="text-[10px] font-bold text-gray-600 mt-0.5 flex items-center gap-1">
                              <Calendar className="w-2.5 h-2.5" />
                              {new Date(p.date).toLocaleDateString('es-AR',{day:'2-digit',month:'short',year:'numeric'})}
                              <span className="text-gray-700 mx-0.5">·</span>
                              <span>{new Intl.NumberFormat('es-AR',{style:'currency',currency:'ARS'}).format(p.totalPagar)}</span>
                            </p>
                          </div>
                        </div>

                        {/* Progress Bar Area */}
                        <div className="w-full space-y-1.5 pt-1">
                          <div className="flex justify-between items-center">
                            <p className={`text-[10px] font-bold ${isComplete ? 'text-green-500' : 'text-figmaCyan'}`}>
                              {checkedCount} / {totalCount} productos chequeados
                            </p>
                            <p className={`text-[10px] font-black ${isComplete ? 'text-green-500' : 'text-gray-500'}`}>
                              {progress}%
                            </p>
                          </div>
                          <div className="h-1.5 w-full bg-white/5 rounded-full overflow-hidden">
                            <div 
                              className={`h-full rounded-full transition-all duration-500 ${isComplete ? 'bg-green-500' : 'bg-figmaCyan shadow-[0_0_8px_rgba(45,212,191,0.5)]'}`} 
                              style={{ width: `${progress}%` }} 
                            />
                          </div>
                        </div>
                      </button>
                    )})}
                  </div>
                )}
              </div>
            )
          )}

        </main>

        <nav className="absolute bottom-0 inset-x-0 bg-[#0A0C10] border-t border-white/5 px-6 pb-8 pt-4 flex justify-between items-center shrink-0">
            {[
                { id: 'inicio', icon: Home, label: 'Inicio' },
                { id: 'ofertas', icon: Tag, label: 'Ofertas' },
                { id: 'escanear', icon: Scan, label: 'Escanear', action: () => setIsScannerOpen(true) },
                { id: 'changuito', icon: ShoppingCart, label: 'Carrito', badge: cart.length },
                { id: 'pedidos',   icon: ClipboardList, label: 'Pedidos', badge: selectedPedido ? selectedPedido.items.length - (selectedPedido.checkedEans || []).length : pedidos.length }
            ].map(tab => (
                <button
                    key={tab.id}
                    onClick={tab.action ? tab.action : () => setActiveTab(tab.id)}
                    className={`flex flex-col items-center gap-1.5 transition-all ${activeTab === tab.id ? 'text-figmaCyan' : 'text-gray-600'}`}
                >
                    <div className="relative">
                        <tab.icon className={`w-5 h-5 ${activeTab === tab.id ? 'text-figmaCyan' : ''}`} />
                        {tab.badge > 0 && <span className="absolute -top-1.5 -right-2 bg-figmaRed text-white text-[8px] font-black w-4 h-4 rounded-full flex items-center justify-center border border-[#0A0C10]">{tab.badge}</span>}
                    </div>
                    <span className="text-[9px] font-bold uppercase tracking-tighter">{tab.label}</span>
                </button>
            ))}
        </nav>
      </div>

      {isScannerOpen && (
        <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/95 backdrop-blur-3xl animate-fade-in">
          <div className="relative w-full max-w-[440px] h-full sm:h-auto"><Scanner onScanSuccess={handleBarcodeScanned} onScanError={alert} onClose={() => setIsScannerOpen(false)} /></div>
        </div>
      )}

      <ProductBottomSheet isOpen={isBottomSheetOpen} product={scannedProduct} getEffectivePrice={getEffectivePrice} onClose={() => setIsBottomSheetOpen(false)} onConfirm={handleConfirmProduct} />
      <ManualFallbackModal isOpen={isManualModalOpen} ean={activeEan} errorType={errorType} onClose={() => setIsManualModalOpen(false)} onConfirm={(p) => { 
        setCart(prev => [...prev, p]); 
        setIsManualModalOpen(false); 
        if (selectedPedido) setActiveTab('pedidos');
      }} />

      {/* ── MODAL DE CONFIRMACIÓN: VACIAR CARRITO ─────────────────────── */}
      {showClearConfirm && (
        <div className="fixed inset-0 z-[300] flex items-end justify-center animate-fade-in" onClick={() => setShowClearConfirm(false)}>
          <div className="absolute inset-0 bg-black/70 backdrop-blur-md" />

          <div
            className="relative w-full max-w-[440px] bg-[#0F1115] border border-white/8 rounded-t-[3rem] px-8 pt-8 pb-12 flex flex-col items-center gap-6 shadow-2xl animate-slide-up"
            onClick={e => e.stopPropagation()}
          >
            {/* Pastilla drag handle */}
            <div className="w-10 h-1 bg-white/15 rounded-full -mt-2 mb-2" />

            {/* Icono */}
            <div className="w-16 h-16 rounded-[1.5rem] bg-red-950/50 border border-red-900/40 flex items-center justify-center">
              <Trash2 className="w-7 h-7 text-red-500" />
            </div>

            {/* Texto */}
            <div className="text-center space-y-1.5">
              <h3 className="text-lg font-black text-white tracking-tight">¿Vaciar el changuito?</h3>
              <p className="text-xs font-medium text-gray-500 leading-relaxed max-w-[260px]">
                Se eliminarán todos los productos que agregaste. Esta acción no se puede deshacer.
              </p>
            </div>

            {/* Botones */}
            <div className="w-full flex flex-col gap-3 pt-2">
              <button
                onClick={() => { setCart([]); setShowClearConfirm(false); }}
                className="w-full py-4 bg-figmaRed text-white rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all shadow-lg shadow-red-900/30 flex items-center justify-center gap-2"
              >
                <Trash2 className="w-4 h-4" />
                Sí, vaciar
              </button>
              <button
                onClick={() => setShowClearConfirm(false)}
                className="w-full py-4 bg-white/5 text-gray-400 rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all border border-white/5"
              >
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}
      {/* ── MODAL: GUARDAR PEDIDO ───────────────────────────────────── */}
      {showSaveOrderModal && (
        <div className="fixed inset-0 z-[300] flex items-end justify-center animate-fade-in" onClick={() => setShowSaveOrderModal(false)}>
          <div className="absolute inset-0 bg-black/70 backdrop-blur-md" />
          <div className="relative w-full max-w-[440px] bg-[#0F1115] border border-white/8 rounded-t-[3rem] px-8 pt-8 pb-14 flex flex-col items-center gap-5 shadow-2xl animate-slide-up" onClick={e => e.stopPropagation()}>
            <div className="w-10 h-1 bg-white/15 rounded-full -mt-2" />
            <div className="w-16 h-16 rounded-[1.5rem] bg-figmaPurple/20 border border-figmaPurple/30 flex items-center justify-center">
              <ClipboardList className="w-7 h-7 text-figmaPurple" />
            </div>
            <div className="text-center space-y-1">
              <h3 className="text-lg font-black text-white tracking-tight">¿Guardar este pedido?</h3>
              <p className="text-xs text-gray-500">Asigná un nombre para identificarlo</p>
            </div>

            {/* NOMBRES RÁPIDOS */}
            <div className="flex flex-wrap gap-2 justify-center">
              {['Camila', 'Isabela', 'Nora', 'Gustavo'].map(n => (
                <button key={n} onClick={() => setSaveOrderName(n)}
                  className={`px-4 py-1.5 rounded-full text-xs font-black uppercase tracking-wide active:scale-95 transition-all border ${
                    saveOrderName === n
                      ? 'bg-figmaPurple text-white border-figmaPurple'
                      : 'bg-white/5 text-gray-400 border-white/10 hover:bg-white/10 hover:text-white'
                  }`}>
                  {n}
                </button>
              ))}
            </div>

            {/* INPUT LIBRE */}
            <input
              type="text"
              value={saveOrderName}
              onChange={e => setSaveOrderName(e.target.value)}
              placeholder="O escribá otro nombre..."
              className="w-full bg-white/5 border border-white/10 rounded-2xl px-5 py-4 text-white font-bold text-sm placeholder-gray-700 focus:outline-none focus:border-figmaPurple/50 transition-colors"
              onKeyDown={e => e.key === 'Enter' && handleSaveOrder()}
            />

            <div className="w-full flex flex-col gap-3 pt-1">
              <button
                onClick={handleSaveOrder}
                disabled={!saveOrderName.trim()}
                className="w-full py-4 bg-figmaRed disabled:opacity-30 text-white rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all flex items-center justify-center gap-2"
              >
                <ClipboardList className="w-4 h-4" />
                Guardar Pedido
              </button>
              <button onClick={() => setShowSaveOrderModal(false)}
                className="w-full py-4 bg-white/5 text-gray-400 rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all border border-white/5">
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}

      {/* ── MODAL: NUEVA LISTA VACÍA ───────────────────────────────────── */}
      {showCreateListModal && (
        <div className="fixed inset-0 z-[300] flex items-end justify-center animate-fade-in" onClick={() => setShowCreateListModal(false)}>
          <div className="absolute inset-0 bg-black/70 backdrop-blur-md" />
          <div className="relative w-full max-w-[440px] bg-[#0F1115] border border-white/8 rounded-t-[3rem] px-8 pt-8 pb-14 flex flex-col items-center gap-5 shadow-2xl animate-slide-up" onClick={e => e.stopPropagation()}>
            <div className="w-10 h-1 bg-white/15 rounded-full -mt-2" />
            <div className="w-16 h-16 rounded-[1.5rem] bg-figmaPurple/20 border border-figmaPurple/30 flex items-center justify-center">
              <ClipboardList className="w-7 h-7 text-figmaPurple" />
            </div>
            <div className="text-center space-y-1">
              <h3 className="text-lg font-black text-white tracking-tight">Crear Lista Nueva</h3>
              <p className="text-xs text-gray-500">Ideal para usar como recordatorio en la tienda</p>
            </div>

            <input
              autoFocus
              type="text"
              value={createListName}
              onChange={e => setCreateListName(e.target.value)}
              placeholder="Ej: Compras del finde..."
              className="w-full bg-white/5 border border-white/10 rounded-2xl px-5 py-4 text-white font-bold text-sm placeholder-gray-700 focus:outline-none focus:border-figmaPurple/50 transition-colors"
              onKeyDown={e => e.key === 'Enter' && handleCreateEmptyList()}
            />

            <div className="w-full flex flex-col gap-3 pt-1">
              <button
                onClick={handleCreateEmptyList}
                disabled={!createListName.trim()}
                className="w-full py-4 bg-figmaPurple disabled:opacity-30 text-white rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all flex items-center justify-center gap-2"
              >
                <Plus className="w-4 h-4" />
                Crear Lista
              </button>
              <button onClick={() => setShowCreateListModal(false)}
                className="w-full py-4 bg-white/5 text-gray-400 rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all border border-white/5">
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}

      {/* ── MODAL: REEMPLAZO MANUAL ───────────────────────────────────── */}
      {showReplacementModal && replacementProduct && selectedPedido && (
        <div className="fixed inset-0 z-[300] flex items-end justify-center animate-fade-in" onClick={() => setShowReplacementModal(false)}>
          <div className="absolute inset-0 bg-black/70 backdrop-blur-md" />
          <div className="relative w-full max-w-[440px] bg-[#0F1115] border border-white/8 rounded-t-[3rem] px-8 pt-8 pb-14 flex flex-col items-center gap-5 shadow-2xl animate-slide-up" onClick={e => e.stopPropagation()}>
            <div className="w-10 h-1 bg-white/15 rounded-full -mt-2" />
            <div className="w-16 h-16 rounded-[1.5rem] bg-figmaCyan/20 border border-figmaCyan/30 flex items-center justify-center p-2">
              {replacementProduct.imageUrl ? (
                <img src={replacementProduct.imageUrl} alt="" className="w-full h-full object-contain" />
              ) : (
                <Package className="w-7 h-7 text-figmaCyan" />
              )}
            </div>
            <div className="text-center space-y-1">
              <h3 className="text-lg font-black text-white tracking-tight leading-tight">Escaneaste<br/><span className="text-figmaCyan">{replacementProduct.name}</span></h3>
              <p className="text-xs text-gray-400 max-w-[280px] pt-1">¿Este producto reemplaza a alguno de tus ítems manuales pendientes?</p>
            </div>

            <div className="w-full flex flex-col gap-2 pt-2 max-h-[30vh] overflow-y-auto no-scrollbar">
              {selectedPedido.items
                .filter(i => i.isManual && !(selectedPedido.checkedEans || []).includes(i.ean))
                .sort((a, b) => {
                  const score = (str) => {
                    const norm = s => s.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "").replace(/[^a-z0-9\s]/g, "");
                    const w1 = norm(str).split(/\s+/).filter(Boolean);
                    const w2 = norm(replacementProduct.name).split(/\s+/).filter(Boolean);
                    let inter = 0;
                    for (let w of w1) { if (w2.includes(w)) inter++; }
                    let s = inter * 10;
                    if (norm(replacementProduct.name).includes(norm(str)) || norm(str).includes(norm(replacementProduct.name))) s += 5;
                    return s;
                  };
                  return score(b.name) - score(a.name);
                })
                .map(item => (
                  <button
                    key={item.ean}
                    onClick={() => handleReplacementSelect(item.ean)}
                    className="w-full py-4 bg-[#1A1D24] text-white rounded-2xl font-black text-sm active:scale-95 transition-all border border-white/10 flex items-center gap-3 px-4"
                  >
                    <ClipboardList className="w-5 h-5 text-figmaPurple shrink-0" />
                    <span className="truncate flex-1 text-left">Reemplazar "{item.name}"</span>
                  </button>
                ))}
            </div>

            <div className="w-full flex flex-col gap-3 pt-2">
              <button
                onClick={() => handleReplacementSelect(null)}
                className="w-full py-4 bg-figmaRed text-white rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all shadow-lg shadow-red-900/30"
              >
                Ninguno, agregar nuevo
              </button>
              <button onClick={() => setShowReplacementModal(false)}
                className="w-full py-4 bg-white/5 text-gray-400 rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all border border-white/5">
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}

      {/* ── MODAL: CONFIRMAR DICTADO POR VOZ ──────────────────────────── */}
      {pendingVoiceItems !== null && (
        <div className="fixed inset-0 z-[300] flex items-end justify-center animate-fade-in" onClick={() => setPendingVoiceItems(null)}>
          <div className="absolute inset-0 bg-black/70 backdrop-blur-md" />
          <div className="relative w-full max-w-[440px] bg-[#0F1115] border border-white/8 rounded-t-[3rem] px-8 pt-8 pb-14 flex flex-col items-center gap-5 shadow-2xl animate-slide-up" onClick={e => e.stopPropagation()}>
            <div className="w-10 h-1 bg-white/15 rounded-full -mt-2" />
            <div className="w-16 h-16 rounded-[1.5rem] bg-figmaPurple/20 border border-figmaPurple/30 flex items-center justify-center">
              <Mic className="w-7 h-7 text-figmaPurple" />
            </div>
            <div className="text-center space-y-1">
              <h3 className="text-lg font-black text-white tracking-tight">Revisar Dictado</h3>
              <p className="text-xs text-gray-400 max-w-[280px]">Revisá lo que entendimos antes de agregarlo a la lista. Podés eliminar lo que sobre.</p>
            </div>

            <div className="w-full flex flex-col gap-2 pt-2 max-h-[35vh] overflow-y-auto no-scrollbar">
              {pendingVoiceItems.length === 0 ? (
                <p className="text-center text-sm font-bold text-gray-500 py-4">No quedó ningún ítem.</p>
              ) : (
                pendingVoiceItems.map((item, index) => (
                  <div key={index} className="w-full py-3 bg-[#1A1D24] rounded-2xl border border-white/10 flex items-center justify-between px-4">
                    <span className="text-sm font-bold text-white truncate pr-2">{item}</span>
                    <button onClick={() => removePendingVoiceItem(index)} className="p-2 text-gray-400 active:scale-95 transition-all bg-white/5 rounded-xl">
                      <Trash2 className="w-4 h-4 text-figmaRed" />
                    </button>
                  </div>
                ))
              )}
            </div>

            <div className="w-full flex flex-col gap-3 pt-2">
              <button
                onClick={confirmVoiceItems}
                disabled={pendingVoiceItems.length === 0}
                className="w-full py-4 bg-figmaPurple disabled:opacity-30 text-white rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all flex items-center justify-center gap-2 shadow-lg shadow-purple-900/30"
              >
                <Check className="w-5 h-5" />
                Agregar a la Lista
              </button>
              <button onClick={() => setPendingVoiceItems(null)}
                className="w-full py-4 bg-white/5 text-gray-400 rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all border border-white/5">
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}

    </div>
  );
}


      {/* ── MODAL: VINCULAR A CARRITO ───────────────────────────────────── */}
      {itemToLink && selectedPedido && (
        <div className="fixed inset-0 z-[300] flex items-end justify-center animate-fade-in" onClick={() => setItemToLink(null)}>
          <div className="absolute inset-0 bg-black/70 backdrop-blur-md" />
          <div className="relative w-full max-w-[440px] bg-[#0F1115] border border-white/8 rounded-t-[3rem] px-8 pt-8 pb-14 flex flex-col items-center gap-5 shadow-2xl animate-slide-up" onClick={e => e.stopPropagation()}>
            <div className="w-10 h-1 bg-white/15 rounded-full -mt-2" />
            <div className="w-16 h-16 rounded-[1.5rem] bg-figmaCyan/20 border border-figmaCyan/30 flex items-center justify-center p-2">
              <ClipboardList className="w-7 h-7 text-figmaCyan" />
            </div>
            <div className="text-center space-y-1">
              <h3 className="text-lg font-black text-white tracking-tight leading-tight">¿Vincular producto?</h3>
              <p className="text-xs text-gray-400 max-w-[280px] pt-1">¿Querés relacionar <span className="text-figmaCyan font-bold">"{itemToLink.name}"</span> con algún producto de tu changuito?</p>
            </div>

            <div className="w-full flex flex-col gap-2 pt-2 max-h-[30vh] overflow-y-auto no-scrollbar">
                          return {
                            ...i,
                            ean: cartItem.ean,
                            name: cartItem.name,
                            isManual: false,
                            listPrice: cartItem.listPrice,
                            price: cartItem.price,
                            tipoOferta: cartItem.tipoOferta,
                            qtyPromoFormula: cartItem.qtyPromoFormula,
                            ofertaActiva: cartItem.ofertaActiva,
                            imageUrl: cartItem.imageUrl
                          };
                        }
                        return i;
                      });
                      
                      const c = new Set(p.checkedEans || []);
                      c.delete(itemToLink.ean);
                      c.add(cartItem.ean);

                      const tp = newItems.filter(i => !i.isManual).reduce((acc, i) => acc + getEffectivePrice(i) * i.quantity, 0);
                      const ts = newItems.filter(i => !i.isManual).reduce((acc, i) => acc + i.listPrice * i.quantity, 0);

                      return { 
                        ...p, 
                        items: newItems, 
                        checkedEans: Array.from(c),
                        totalPagar: tp,
                        totalAhorro: ts - tp
                      };
                    }));
                    setItemToLink(null);
                  }}
                  className="w-full py-3 bg-[#1A1D24] text-white rounded-2xl font-black text-sm active:scale-95 transition-all border border-white/10 flex items-center gap-3 px-4"
                >
                  <div className={`w-8 h-8 rounded-lg overflow-hidden flex items-center justify-center shrink-0 bg-white p-1`}>
                    {cartItem.imageUrl ? (
                      <img src={cartItem.imageUrl} alt="" className="w-full h-full object-contain" />
                    ) : (
                      <Package className="w-5 h-5 text-gray-300" />
                    )}
                  </div>
                  <div className="flex-1 min-w-0 text-left">
                    <p className="truncate text-sm">{cartItem.name}</p>
                    <p className="text-[10px] text-gray-400">{new Intl.NumberFormat('es-AR',{style:'currency',currency:'ARS'}).format(getEffectivePrice(cartItem))}</p>
                  </div>
                </button>
              ))}
            </div>

            <div className="w-full flex flex-col gap-3 pt-2">
              <button
                onClick={() => {
                  setPedidos(prev => prev.map(p => {
                    if (p.id !== selectedPedidoId) return p;
                    const c = new Set(p.checkedEans || []);
                    c.add(itemToLink.ean);
                    return { ...p, checkedEans: Array.from(c) };
                  }));
                  setItemToLink(null);
                }}
                className="w-full py-4 bg-figmaCyan text-black rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all shadow-lg shadow-cyan-900/30"
              >
                Solo marcar como chequeado
              </button>
              <button onClick={() => setItemToLink(null)}
                className="w-full py-4 bg-white/5 text-gray-400 rounded-2xl font-black text-sm uppercase tracking-widest active:scale-95 transition-all border border-white/5">
                Cancelar
              </button>
            </div>
          </div>
        </div>
      )}

    </div>
  );
}
